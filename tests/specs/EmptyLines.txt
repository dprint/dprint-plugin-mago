== should add empty line after opening tag (default) ==
<?php
echo 'hello';

[expect]
<?php

== should add empty line after namespace (default) ==
<?php

[expect]
<?php

namespace App;

class Foo {}

== should add empty line after use statements (default) ==
<?php
namespace App;
use Other\Foo;
class Bar {}

[expect]
<?php

namespace App;

use Other\Foo;

class Bar {}

== should add empty line after method (default) ==
<?php
class Foo {
    public function bar() { return 1; }
    public function baz() { return 2; }
}

[expect]
<?php

class Foo
{
    public function bar()
    {
        return 1;
    }

    public function baz()
    {
        return 2;
    }
}

    return $x;
}
