== should format basic echo ==
<?php
echo   'hello'  ;

[expect]
<?php

echo 'hello';

== should format class ==
<?php
class Foo{public function bar(){return 1;}}

[expect]
<?php

class Foo
{
    public function bar()
    {
        return 1;
    }
}

== should format function ==
<?php
function test($a,$b){return $a+$b;}

[expect]
<?php

function test($a, $b)
{
    return $a + $b;
}

== should format if statement ==
<?php
if($a==1){echo 'one';}else{echo 'other';}

[expect]
<?php

if ($a == 1) {
    echo 'one';
} else {
    echo 'other';
}

== should format namespace and use ==
<?php
namespace App\Models;use Illuminate\Database\Eloquent\Model;class User extends Model{public function name(){return $this->name;}}

[expect]
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

class User extends Model
{
    public function name()
    {
        return $this->name;
    }
}

== should format closure ==
<?php

[expect]
<?php

$fn = function ($x) use ($y) {
    return $x + $y;
};

    echo $i;
}

== should format while loop ==

== should format switch statement ==
<?php
switch($x){case 1:echo 'one';break;default:echo 'other';}

[expect]
<?php

switch ($x) {
    case 1:
        echo 'one';
        break;
    default:
        echo 'other';
}

== should format try catch ==

[expect]

try {
    doSomething();
} catch (Exception $e) {
    log($e);
} finally {
    cleanup();
}
